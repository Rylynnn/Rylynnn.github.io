[原题](http://codeforces.com/contest/721/problem/D)


## 题意

给你一个数列，你可以要进行k次操作，每次选一个数+x或者-x，使得最后所有数的乘积最小

## 分析

负数的乘积最小啦！
所以如果有偶数个负数，那么一定需要将一个数变成负数！
如果有奇数个负数，增加绝对值就行啦！
优先队列维护每次需要修改的值（绝对值最小的那个）

```
/*
wa:wa在了ll上面。。。
*/
#include <bits/stdc++.h>
#define ll long long
using namespace std;
struct node{
    ll b;
    ll c;
    ll num;
    bool operator < (const node n)const{
        return c>n.c;
    }
}a,arr[200009];
bool cmp(node p,node q){
    return p.num<q.num;
}
priority_queue<node> q;
int main()
{
    ll n,k,x;
    while(scanf("%I64d%I64d%I64d",&n,&k,&x)!=EOF){
        ll num=0;
        for(ll i=0;i<n;i++){
            scanf("%I64d",&a.b);
            if(a.b<0){
                num++;
            }
            a.c=abs(a.b);
            a.num=i;
            q.push(a);
        }
        while(k--){
            if(num&1){
                if(q.top().b<0){
                    a=q.top();
                    q.pop();
                    a.b-=x;
                    a.c=abs(a.b);
                    q.push(a);
                }
                else{
                    a=q.top();
                    q.pop();
                    a.b+=x;
                    a.c=abs(a.b);
                    q.push(a);
                }
            }
            else{
                if(q.top().b<0){
                    a=q.top();
                    q.pop();
                    a.b+=x;
                    if(a.b>=0){
                        num--;
                    }
                    a.c=abs(a.b);
                    q.push(a);
                }
                else{
                    a=q.top();
                    q.pop();
                    a.b-=x;
                    if(a.b<0){
                        num++;
                    }
                    a.c=abs(a.b);
                    q.push(a);
                }
            }
        }
        ll i=0;
        while(!q.empty()){
            arr[i++]=q.top();
            q.pop();
        }
        sort(arr,arr+n,cmp);
        printf("%I64d",arr[0].b);
        for(ll i=1;i<n;i++){
            printf(" %I64d",arr[i].b);
        }
        cout<<endl;
    }
}
```
